# Radio

Form Radio input component with a label wrapper

#### Default

<Preview>
  <Story name="default">
    {({ id }) => {
      const [value, setValue] = useState('option1');
      return (
        <Box>
          <Radio
            mb={2}
            name={id('rd-default')}
            label="Option 1"
            value="option1"
            checked={value === 'option1'}
            onChange={e => {
              setValue(e.target.value);
            }}
          />
          <Radio
            name={id('rd-default')}

    }}
  </Story>
</Preview>
