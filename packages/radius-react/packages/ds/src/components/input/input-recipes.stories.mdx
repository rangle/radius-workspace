### Search Field

export function Search() {
  const [query, setQuery] = useState(null);
  return (
    <Input
      placeholder="Search"
      value={query}
      onChange={e => setQuery(e.target.value)}
      leftIcon={<Icons.Search />}
      rightIcon={query && <Icons.Close color="brand.primary" />}
    />
  );
}

<Preview>
  <Story name="search field">
    {() => {
      const [query, setQuery] = useState(null);
      return (
        <Input
          placeholder="Search"
          value={query}
          onChange={e => setQuery(e.target.value)}
          leftIcon={<Icons.Search aria-hidden />}
          rightIcon={
            query && <Icons.Close title="clear" color="brand.primary" />
          }
        />
      );
    }}
  </Story>
</Preview>

### Visibility Toggle

<Preview>
  <Story name="visibility toggle">
    {() => {
      const [visibility, setVisibility] = useState(false);
      return (
        <Input
          type={visibility ? 'password' : 'text'}
          value="password"
          rightIcon={
            visibility ? (
              <Icons.Visibility
                title="toggle visibility"
                onClick={() => setVisibility(false)}
                color="brand.primary"
              />
            ) : (
              <Icons.VisibilityOff
                title="toggle visibility"
                onClick={() => setVisibility(true)}
                color="brand.primary"
              />
            )
          }
          mb={3}
        />
      );
    }}
  </Story>
</Preview>
